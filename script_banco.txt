CREATE TABLE TB_ESTACIONAMENTO_GS(CD NUMBER(20) PRIMARY KEY NOT NULL, LOCAL VARCHAR(255) NOT NULL);

CREATE TABLE TB_CARRO_GS(ID NUMBER(20) PRIMARY KEY NOT NULL, PLACA VARCHAR(255) NOT NULL, 
NOME_PESSOA VARCHAR(255) NOT NULL, MODELO_CARRO VARCHAR(255) NOT NULL, 
DATA_ESTACIONAMENTO VARCHAR(255) NOT NULL, ESTACIONAMENTO_CODIGO NUMBER(20) REFERENCES TB_ESTACIONAMENTO_GS(CD) NOT NULL);


CREATE SEQUENCE TB_ESTACIONAMENTO INCREMENT BY 1;

CREATE SEQUENCE TB_CARRO INCREMENT BY 1;
